cmake_minimum_required(VERSION 2.6)
IF (DEFINED ENV{CHAOS_BUNDLE} )
  include($ENV{CHAOS_BUNDLE}/tools/project_template/CMakeChaos.txt)
ELSE()
  MESSAGE(STATUS "STANDALONE, environment CHAOS_BUNDLE defined")
  IF (NOT CMAKE_INSTALL_PREFIX) 
    MESSAGE(fatal "you must add -DCMAKE_INSTALL_PREFIX=<dir>")
  ELSE()
    MESSAGE(STATUS "install prefix ${CMAKE_INSTALL_PREFIX}")
  ENDIF()
  INCLUDE_DIRECTORIES(. ../..)
  LINK_DIRECTORIES(${CMAKE_BINARY_DIR}/build ${CMAKE_INSTALL_PREFIX}/lib)
ENDIF()



SET(debug_src source/debug.cpp )
SET(debug_test_src source/main.cpp )

IF(FrameworkLib)
SET(libs chaos_common ${FrameworkLib})
ELSE()
SET(libs pthread)
ENDIF()
IF(BUILD_FORCE_STATIC)
ADD_LIBRARY(common_debug STATIC ${debug_src})
ADD_EXECUTABLE(test_debug ${debug_test_src} ${debug_src})
TARGET_LINK_LIBRARIES(test_debug ${libs})
ELSE()
ADD_LIBRARY(common_debug SHARED ${debug_src})
ADD_EXECUTABLE(test_debug ${debug_test_src})
TARGET_LINK_LIBRARIES(test_debug common_debug ${libs})
ENDIF()

FILE(GLOB headers "source/*.h")
INSTALL(FILES ${headers} DESTINATION include/common/debug)
INSTALL_TARGETS(/lib  common_debug)
INSTALL_TARGETS(/bin test_debug)
